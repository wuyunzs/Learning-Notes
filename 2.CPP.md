引用语法：
作用：取代指针，让指针操作更简洁

引用和指针有什么区别：

 1.没有空引用

 3.没有二级引用
 4.引用初始化完成就不能修改（像指针常量)

 const引用：可以初始化为常量或者函数常量传递

    n2 = tmp;

	1.无参宏不带数据类型

		return -1;
	}
	memcpy(pBuf, m_pBuf, m_nLen + 1);
	int nMaxCount = m_nLen - strlen(pStr) + 1;
	int nCount = 0;
	int nState = 0;
	int nNoTarget = 0;

	for (size_t i = 0; i < nMaxCount; i++)
	{
		nNoTarget = memcmp(pBuf + i, pStr, strlen(pStr));
		if (nNoTarget && !nState)//匹配失败，否
		{
			nState = 1;
			*(size_t*)((int*)*ppPosi + nCount) = (size_t)&pBuf[i];
			nCount++;
		}
		if (!nNoTarget && nState)//匹配成功，是
		{
			nState = 0;
		}
		if (!nNoTarget)//匹配成功
		{
			pBuf[i] = '\0';
		}
	}

	return nCount;
}

		//下一位置					坐标
		if (x + range + 1 >= m_nWidth - 1) {
			return false;
		}

		//触碰
		for (size_t i = 0; i < 4; i++)
		{
			if (m_pMap[y * range * i + x + 4] == 1) {
				return false;
			}
		}
		return true;

	case 'S':

		//自身范围
		for (size_t i = h - 1; i > 0; i--)
		{
			for (size_t j = 0; j < w; j++)
			{
				if (*(psz + i * w + j) == 1)
				{
					range = i;
					goto retRangeS;
				}
			}
		}
	retRangeS:
		if (y + range + 1 >= m_nHeight - 1) {
			return false;
		}

		for (size_t i = 0; i < 4; i++)
		{
			if (m_pMap[(y + 4) * w + x + i] == 1) {
				return false;
			}
		}
		return true;
	case 'A':

		for (size_t i = 0; i < w; i++)
		{
			for (size_t j = 0; j < h; j++)
			{
				if (*(psz + j * w + i) == 1)
				{
					range = i;
					goto retRangeA;
				}
			}
		}
	retRangeA:
		if (x + range - 1 <= 0) {
			return false;
		}
		for (size_t i = 0; i < 4; i++)
		{
			if (m_pMap[y * w * i + x - 1] == 1) {
				return false;
			}
		}
		return true;
	default:
		break;
	}
	return false;
}

					score += removed * 100 * level;
					if (lines >= level * 10)

					printf("Press any key to exit...\n");
					_getch();
					delete pBlock;
					return;
				}
				graphical.Draw(pBlock);
			}
			lastFallTime = GetTickCount();
		}

	const char* data = pBlock->GetData();

	{
		bool full = true;

		{
			if (m_pMap[i * m_nWidth + j] != 1)
			{
				full = false;
				break;
			}
		}
		if (full)
		{
			for (int k = i; k >= 2; k--)
			{
				for (int j = 1; j < m_nWidth - 1; j++)
				{
					m_pMap[k * m_nWidth + j] = m_pMap[(k - 1) * m_nWidth + j];

			}
			for (int j = 1; j < m_nWidth - 1; j++)
			{
				m_pMap[1 * m_nWidth + j] = 0;
			}
			removed++;
			i++;
		}
	}
	return removed;
}
