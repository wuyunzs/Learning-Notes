# 原码补码反码

**只有负数才有原码，反码和补码**

**原码：最高位为符号位，其余各位为数值本身的绝对值**

**反码：符号位不变，其余各位取反**

**补码：反码+1**

**减法采用的是补码形式**

# 单位转换

1Byte = 8Bit，一字节=八位

1024Byte = 1KB

1024KB = 1MB

1024MB = 1GB

1024GB = 1TB

以下比较少遇到

1024TB = 1PB

1024PB = 1EB

1024EB = 1ZB

所以

1TB = 2^10GB = 2^20MB = 2^30KB = 2^40Byte

```
#include <stdio.h>
#include <math.h>

//for循环效率更高
int ForFib(int nItem)
{
	if (nItem <= 0)
	{
		return -1;
	}

	if (nItem == 1 ||nItem == 2)
	{
		return 1;
	}
	int nTmp = nItem - 2;
	int nFib1 = 1;
	int nFib2 = 1;
	for (int i = 0; i < nTmp; i++)
	{
		int nK = nFib1 + nFib2;
		nFib1 = nFib2;
		nFib2 = nK;

	}
	return nFib2;
	
}
//递归效率低
int RecursionFib(int nItem)
{
	if (nItem <= 0)
	{
		return -1;
	}
	if (nItem == 1 || nItem == 2)
	{
		return 1;
	}
	return RecursionFib(nItem-1) + RecursionFib(nItem - 2);
}

int main(int argc, char* argv[])
{
	
	for (size_t i = 1; i < 45; i++)
	{
		printf("%d\r\n", ForFib(i));
	}
	printf("============================================\r\n");
	for (size_t i = 1; i < 45; i++)
	{
		printf("%d\r\n", RecursionFib(i));
	}
	
	getchar();
	return 0;
}

# 调用约定

```
/*
当调用函数时，将发生以下行为
    1，按调用约定传递参数。（暂时只介绍三个）
    1.1调用约定：调用方(caller)需要和被调用方(callee)作出以下约定：
        1.参数的传递方向
        2.参数的传输媒介
        3.函数返回值的位置
        4.释放参数空间的负责方，有且仅有一方负责释放参数空间。
        
        __cdecl:参数使用栈空间传递，从右往左，
                函数返回值在寄存器，由调用方负责释放参数空间
        
        __stdcall:参数使用栈空间传递，从右往左，
                  函数返回值在寄存器，由被调用方负责释放参数空间
        
        __fastcall:左数前两个参数使用寄存器传递，其它参数使用栈空间传递，
                   从右往左，函数返回值在寄存器，由被调用方负责释放参数空间
        (__cdecl,__stdcall最常用且可跨平台，__fastcall是微软独有的)
    2，保存返回地址。
    3，保存调用方的栈信息(栈底)。
    4，更新当前栈底到栈顶。(把当前栈顶作为被调方的栈底)
    5，为局部变量申请空间。(抬高栈顶)
    6，保存寄存器环境。(把即将使用的寄存器原值保存到栈里)
    *7，如果编译选项有/ZI /Zi，则将局部变量初始化为0xCCCCCCCC
    8，执行函数体
    9，恢复寄存器环境
    10，释放局部变量的空间
    11，恢复调用方的栈信息。(栈底)
    12A，__cedcl约定，取出当前栈顶作为返回的流程地址，
        返回到调用方后，由调用方清理参数空间。
    12B，_其它约定，取出当前栈顶作为返回的流程地址，
        同时由被调方清理参数空间后才返回。
*/
```

#include <stdio.h>
#include <stdlib.h>
//属于对数阶，log2(n)
int BinarySearch(int ary[], int nMax, int target)
{
	int nLeft = 0;
	int nRight = nMax;
	int nMiddle = (nRight + nLeft) / 2;
	while (nRight > nLeft)
	{
		if (ary[nMiddle] > target)
		{
			nRight = nMiddle - 1;
		}
		if (ary[nMiddle] < target)
		{
			nLeft = nMiddle + 1;
		}
		if (ary[nMiddle] == target)
		{
			return nMiddle;
		}
		nMiddle = (nRight + nLeft) / 2;
	}
	return -1;
}

int main(int argc, char* argv[])

#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <crtdbg.h>

#define SAFE_FREE(p) if(p){free(p);(p)=NULL;}

#define malloc(r) _malloc_dbg(r,_NORMAL_BLOCK,__FILE__,__LINE__)
#endif // _DEBUG



void ErrorProc()
{

}	

int main(int argc, char* argv[], char* envp[])
{
	//多资源使用规范
	//1.引用资源的变量在作用域开始定义，并初始化为错误值
	int* pA = NULL;
	char* pB1 = NULL;
	float* pB2 = NULL;
	double* pC = NULL;

	//2.申请资源后，必须检查资源是否有效，无效则处理错误
	pA = (int*)malloc(sizeof(int));
	if (pA == NULL)
	{
		ErrorProc();
		//3.处理完错误后，转移到统一退出位置(EXIT_PROC)
		goto EXIT_PROC;
	}

	//资源有效则正常使用
	*pA = 999;

EXIT_PROC:
	SAFE_FREE(pC);
	SAFE_FREE(pB2);
	SAFE_FREE(pB1);

	//4.释放资源前，必须检查资源是否有效，无效则不处理
	if (pA != NULL)
	{
		free(pA);
		//5.释放资源后，必须将引用资源的变量重置为错误值

	}

# 位运算

```
#include <stdio.h>
#include <stdlib.h>

#define ADD 1
#define DEL 2
#define EDT 4
#define QUE 8

//xxxxxxxxxxxxxxxxxxxxxx nn rrrr qeda
// a : add
// d : del
// e : edt
// q : que
// r : reserved
// n : no use
// x : no define
struct tagPrivilege
{
	//int add : 2;//占2位
	int add : 1;	//占1位
	int : 5;		//占5位，占位置用的
	int : 0;		//位断，重对齐位断，分开上下两部分数据，新开一个int存储下面三个数据
	int del : 1;	//占1位
	int edt : 1;	//占1位
	int que : 1;	//占1位

};

int MyAbs(int n)
{
	int i = n >> 31;//i得到符号位，要么0要么0xFFFFFFFF
	n = n ^ i;		// n ^ 0 ==> n, n ^ 0xFFFFFFFF ==> ~n
	return n - i;	// n - 0 ==> n, ~n + 1(负数的补码)
}

int main(int argc, char* argv[], char* envp[])
{
	struct tagPrivilege pri = { 0 };
	pri.edt = 1;
	pri.add = 1;

	// 无符号数，高位补0；有符号数,高位补符号位(最高位为符号位)
	// nk == -1 ，因为 int edt : 1;是有符号数且只有一位值为1。
	// 所以nk高位补1。nk == 0xFFFFFFFF 即-1
	//如果是 unsigned int edt : 1;那么 nk == 1;

	int nk = pri.edt;
	printf("%d\r\n", nk);

	if (pri.add)
	{
		puts("add");
	}
	if (pri.del)
	{
		puts("del");
	}
	if (pri.edt)
	{
		puts("edt");
	}
	if (pri.que)
	{
		puts("que");
	}


	int nPrivilege = DEL | EDT;//增加权限

	if (nPrivilege & ADD)
	{
		puts("ADD");
	}
	if (nPrivilege & DEL)
	{
		puts("DEL");
	}
	if (nPrivilege & EDT)
	{
		puts("EDT");
	}
	if (nPrivilege & QUE)
	{
		puts("QUE");
	}

	int n = MyAbs(0);
	printf("%d\r\n", n);
	return 0;
}
